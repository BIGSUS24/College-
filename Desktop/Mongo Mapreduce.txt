// Step 1: Create Collection
db.createCollection("Order");

// Step 2: Insert Documents
db.Order.insertMany([
  { Customer_id: 'A1', Amount: 400, Status: 'P' },
  { Customer_id: 'B1', Amount: 300, Status: 'D' },
  { Customer_id: 'A1', Amount: 200, Status: 'F' },
  { Customer_id: 'C1', Amount: 200, Status: 'F' },
  { Customer_id: 'B1', Amount: 700, Status: 'P' },
  { Customer_id: 'B1', Amount: 800, Status: 'P' }
]);

// Step 3: Display All Documents
print("All Documents:");
db.Order.find().forEach(doc => printjson(doc));

// Step 4: MapReduce – Sum of Amount for Each Customer (Status = 'P')
var mapSum = function() {
  emit(this.Customer_id, this.Amount);
};

var reduceSum = function(key, values) {
  return Array.sum(values);
};

db.Order.mapReduce(mapSum, reduceSum, {
  query: { Status: "P" },
  out: "Order_total"
});

print("\nTotal Amount (Status = 'P'):");
db.Order_total.find().forEach(doc => printjson(doc));

// Step 5: MapReduce – Average Amount of Each Customer
var mapAvg = function() {
  emit(this.Customer_id, this.Amount);
};

var reduceAvg = function(key, values) {

};

db.Order.mapReduce(mapAvg, reduceAvg, {
  out: "Order_average"

print("\nAverage Amount of Each Customer:");

var mapMin = function() {
  emit(this.Customer_id, this.Amount);
};

var reduceMin = function(key, values) {
